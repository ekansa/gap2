 15 /**
 16  * @class
 17  * GeoRSS loader factory - inherits from remote loader.
 18  *
 19  * <p>This is a loader class for GeoRSS feeds. Parsing is complicated by the 
 20  * diversity of GeoRSS formats; this parser handles:</p>
 21  * <ul>
 22  *  <li>RSS feeds</li>
 23  *  <li>Atom feeds</li>
 24  * </ul>
 25  * <p>and looks for geographic information in the following formats:</p>
 26  * <ul>
 27  *  <li>GeoRSS-Simple</li>
 28  *  <li>GML </li>
 29  *  <li>W3C Geo</li>
 30  * </ul>
 31  * <p>At the moment, this only supports points; polygons, polylines, and boxes
 32  * will be added at some later point.</p>
 33  *
 34  * @example Usage in TimeMap.init():
 35  
 36     datasets: [
 37         {
 38             title: "GeoRSS Dataset",
 39             type: "georss", // Data to be loaded in GeoRSS
 40             options: {
 41                 url: "mydata.rss" // GeoRSS file to load - must be a local URL
 42             }
 43         }
 44     ]
 45  *
 46  * @param {Object} options          All options for the loader:<pre>
 47  *   {Array} url                        URL of GeoRSS file to load (NB: must be local address)
 48  *   {Function} preloadFunction         Function to call on data before loading
 49  *   {Function} transformFunction       Function to call on individual items before loading
 50  * </pre>
 51  * @return {TimeMap.loaders.remote} Remote loader configured for GeoRSS
 52  */
 53 TimeMap.loaders.georss = function(options) {
 54     var loader = new TimeMap.loaders.remote(options);
 55     loader.parse = TimeMap.loaders.georss.parse;
 56     return loader;
 57 }
 58 
 59 /**
 60  * Static function to parse GeoRSS
 61  *
 62  * @param {XML text} rss        GeoRSS to be parsed
 63  * @return {TimeMapItem Array}  Array of TimeMapItems
 64  */
 65 TimeMap.loaders.georss.parse = function(rss) {
 66     var items = [], data, node, placemarks, pm;
 67     node = GXml.parse(rss);
 68     
 69     // get TimeMap utilty functions
 70     // assigning to variables should compress better
 71     var util = TimeMap.util;
 72     var getTagValue = util.getTagValue,
 73         getNodeList = util.getNodeList,
 74         makePoint = util.makePoint,
 75         makePoly = util.makePoly,
 76         formatDate = util.formatDate,
 77         nsMap = util.nsMap;
 78     
 79     // define namespaces
 80     nsMap.georss = 'http://www.georss.org/georss';
 81     nsMap.gml = 'http://www.opengis.net/gml';
 82     nsMap.geo = 'http://www.w3.org/2003/01/geo/wgs84_pos#';
 83     nsMap.kml = 'http://www.opengis.net/kml/2.2';
 84     
 85     // determine whether this is an Atom feed or an RSS feed
 86     var feedType = (node.firstChild.tagName == 'rss') ? 'rss' : 'atom';
 87     
 88     // look for placemarks
 89     var tName = (feedType == 'rss' ? "item" : "entry");
 90     placemarks = getNodeList(node, tName);
 91     for (var i=0; i<placemarks.length; i++) {
 92         pm = placemarks[i];
 93         data = { options: {} };
 94         // get title & description
 95         data.title = getTagValue(pm, "title");
 96         tName = (feedType == 'rss' ? "description" : "summary");
 97         data.options.description = getTagValue(pm, tName);
 98         // get time information, allowing KML-namespaced time elements
 99         var nList = getNodeList(pm, "TimeStamp", "kml");
100         if (nList.length > 0) {
101             data.start = getTagValue(nList[0], "when", "kml");
102         }
103         // look for timespan
104         if (!data.start) {
105             nList = getNodeList(pm, "TimeSpan", "kml");
106             if (nList.length > 0) {
107                 data.start = getTagValue(nList[0], "begin", "kml");
108                 data.end = getTagValue(nList[0], "end", "kml") ||
109                     // unbounded spans end at the present time
110                     formatDate(new Date());
111             }
112         }
113         // otherwise, use pubDate/updated elements
114         if (!data.start) {
115             if (feedType == 'rss') {
116                 // RSS needs date conversion
117                 var d = new Date(Date.parse(getTagValue(pm, "pubDate")));
118                 // reformat
119                 data.start = formatDate(d);
120             } else {
121                 // atom uses ISO 8601
122                 data.start = getTagValue(pm, "updated");
123             }
124         }
125         // find placemark - single geometry only for the moment
126         PLACEMARK: {
127             var coords, geom;
128             // look for point, GeoRSS-Simple
129             coords = getTagValue(pm, "point", 'georss');
130             if (coords) {
131                 data.point = makePoint(coords); 
132                 break PLACEMARK;
133             }
134             // look for point, GML
135             nList = getNodeList(pm, "Point", 'gml');
136             if (nList.length > 0) {
137                 // GML <pos>
138                 coords = getTagValue(nList[0], "pos", 'gml');
139                 // GML <coordinates>
140                 if (!coords) {
141                     coords = getTagValue(nList[0], "coordinates", 'gml');
142                 }
143                 if (coords) {
144                     data.point = makePoint(coords); 
145                     break PLACEMARK;
146                 }
147             }
148             // look for point, W3C Geo
149             if (getTagValue(pm, "lat", 'geo')) {
150                 coords = [
151                     getTagValue(pm, "lat", 'geo'),
152                     getTagValue(pm, "long", 'geo')
153                 ];
154                 data.point = makePoint(coords); 
155                 break PLACEMARK;
156             }
157             // look for polyline, GeoRSS-Simple
158             coords = getTagValue(pm, "line", 'georss');
159             if (coords) {
160                 data.polyline = makePoly(coords); 
161                 break PLACEMARK;
162             }
163             // look for polygon, GeoRSS-Simple
164             coords = getTagValue(pm, "polygon", 'georss');
165             if (coords) {
166                 data.polygon = makePoly(coords); 
167                 break PLACEMARK;
168             }
169             // look for polyline, GML
170             nList = getNodeList(pm, "LineString", 'gml');
171             if (nList.length > 0) {
172                 geom = "polyline";
173             } else {
174                 nList = getNodeList(pm, "Polygon", 'gml');
175                 if (nList.length > 0) {
176                     geom = "polygon";
177                 }
178             }
179             if (nList.length > 0) {
180                 // GML <posList>
181                 coords = getTagValue(nList[0], "posList", 'gml');
182                 // GML <coordinates>
183                 if (!coords) {
184                     coords = getTagValue(nList[0], "coordinates", 'gml');
185                 }
186                 if (coords) {
187                     data[geom] = makePoly(coords);
188                     break PLACEMARK;
189                 }
190             }
191             
192             // XXX: deal with boxes
193         }
194         items.push(data);
195     }
196     
197     // clean up
198     node = null;
199     placemarks = null;
200     pm = null;
201     nList = null;
202     return items;
203 };
