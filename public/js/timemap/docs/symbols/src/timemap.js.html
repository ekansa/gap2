 32 /**
 33  * @class
 34  * The TimeMap object holds references to timeline, map, and datasets.
 35  * This will create the visible map, but not the timeline, which must be initialized separately.
 36  *
 37  * @constructor
 38  * @param {element} tElement     The timeline element.
 39  * @param {element} mElement     The map element.
 40  * @param {Object} options       A container for optional arguments:<pre>
 41  *   {Boolean} syncBands            Whether to synchronize all bands in timeline
 42  *   {GLatLng} mapCenter            Point for map center
 43  *   {Number} mapZoom               Intial map zoom level
 44  *   {GMapType/String} mapType      The maptype for the map
 45  *   {Array} mapTypes               The set of maptypes available for the map
 46  *   {Function/String} mapFilter    How to hide/show map items depending on timeline state;
 47                                     options: "hidePastFuture", "showMomentOnly", or function
 48  *   {Boolean} showMapTypeCtrl      Whether to display the map type control
 49  *   {Boolean} showMapCtrl          Whether to show map navigation control
 50  *   {Boolean} centerMapOnItems     Whether to center and zoom the map based on loaded item positions
 51  *   {Function} openInfoWindow      Function redefining how info window opens
 52  *   {Function} closeInfoWindow     Function redefining how info window closes
 53  * </pre>
 54  */
 55 function TimeMap(tElement, mElement, options) {
 56     // save elements
 57     
 58     /**
 59      * Map element
 60      * @type DOM Element
 61      */
 62     this.mElement = mElement;
 63     /**
 64      * Timeline element
 65      * @type DOM Element
 66      */
 67     this.tElement = tElement;
 68     
 69     /** 
 70      * Map of datasets 
 71      * @type Object 
 72      */
 73     this.datasets = {};
 74     /**
 75      * Filter chains for this timemap 
 76      * @type Object
 77      */
 78     this.filters = {};
 79     /** 
 80      * Bounds of the map 
 81      * @type GLatLngBounds
 82      */
 83     this.mapBounds = new GLatLngBounds();
 84     
 85     // set defaults for options
 86     
 87     /** 
 88      * Container for optional settings passed in the "options" parameter
 89      * @type Object
 90      */
 91     this.opts = options || {};   // make sure the options object isn't null
 92     // allow map types to be specified by key
 93     if (typeof(options.mapType) == 'string') {
 94         options.mapType = TimeMap.mapTypes[options.mapType];
 95     }
 96     // allow map filters to be specified by key
 97     if (typeof(options.mapFilter) == 'string') {
 98         options.mapFilter = TimeMap.filters[options.mapFilter];
 99     }
100     // these options only needed for map initialization
101     var mapCenter =        options.mapCenter || new GLatLng(0,0),
102         mapZoom =          options.mapZoom || 0,
103         mapType =          options.mapType || G_PHYSICAL_MAP,
104         mapTypes =         options.mapTypes || [G_NORMAL_MAP, G_SATELLITE_MAP, G_PHYSICAL_MAP],
105         showMapTypeCtrl =  ('showMapTypeCtrl' in options) ? options.showMapTypeCtrl : true,
106         showMapCtrl =      ('showMapCtrl' in options) ? options.showMapCtrl : true;
107     
108     // these options need to be saved for later
109     this.opts.syncBands =        ('syncBands' in options) ? options.syncBands : true;
110     this.opts.mapFilter =        options.mapFilter || TimeMap.filters.hidePastFuture;
111     this.opts.centerOnItems =    ('centerMapOnItems' in options) ? options.centerMapOnItems : true;
112     this.opts.theme =            TimeMapTheme.create(options.theme, options);
113     
114     // initialize map
115     if (GBrowserIsCompatible()) {
116         /** 
117          * The associated map object 
118          * @type GMap2
119          */
120         this.map = new GMap2(this.mElement);
121         var map = this.map;
122         if (showMapCtrl) {
123             map.addControl(new GLargeMapControl());
124         }
125         if (showMapTypeCtrl) {
126             map.addControl(new GMapTypeControl());
127         }
128         // drop all existing types
129         var i;
130         for (i=G_DEFAULT_MAP_TYPES.length-1; i>0; i--) {
131             map.removeMapType(G_DEFAULT_MAP_TYPES[i]);
132         }
133         // you can't remove the last maptype, so add a new one first
134         map.addMapType(mapTypes[0]);
135         map.removeMapType(G_DEFAULT_MAP_TYPES[0]);
136         // add the rest of the new types
137         for (i=1; i<mapTypes.length; i++) {
138             map.addMapType(mapTypes[i]);
139         }
140         map.enableDoubleClickZoom();
141         map.enableScrollWheelZoom();
142         map.enableContinuousZoom();
143         // initialize map center and zoom
144         map.setCenter(mapCenter, mapZoom);
145         // must be called after setCenter, for reasons unclear
146         map.setMapType(mapType);
147     }
148 }

156 /**
157  * Intializes a TimeMap.
158  *
159  * <p>This is an attempt to create a general initialization script that will
160  * work in most cases. If you need a more complex initialization, write your
161  * own script instead of using this one.</p>
162  *
163  * <p>The idea here is to throw all of the standard intialization settings into
164  * a large object and then pass it to the TimeMap.init() function. The full
165  * data format is outlined below, but if you leave elements off the script 
166  * will use default settings instead.</p>
167  *
168  * <p>Call TimeMap.init() inside of an onLoad() function (or a jQuery 
169  * $.(document).ready() function, or whatever you prefer). See the examples 
170  * for usage.</p>
171  *
172  * @param {Object} config   Full set of configuration options.
173  *                          See examples/timemapinit_usage.js for format.
174  * @return {TimeMap}        The initialized TimeMap object, for future reference
175  */
176 TimeMap.init = function(config) {
177     
178     // check required elements
179     if (!('mapId' in config) || !config.mapId) {
180         throw "TimeMap.init: No id for map";
181     }
182     if (!('timelineId' in config) || !config.timelineId) {
183         throw "TimeMap.init: No id for timeline";
184     }
185     
186     // set defaults
187     config = config || {}; // make sure the config object isn't null
188     config.options = config.options || {};
189     config.datasets = config.datasets || [];
190     config.bandInfo = config.bandInfo || false;
191     config.scrollTo = config.scrollTo || "earliest";
192     if (!config.bandInfo && !config.bands) {
193         var intervals = config.bandIntervals || 
194             config.options.bandIntervals ||
195             [DT.WEEK, DT.MONTH];
196         // allow intervals to be specified by key
197         if (typeof(intervals) == 'string') {
198             intervals = TimeMap.intervals[intervals];
199         }
200         // save for later reference
201         config.options.bandIntervals = intervals;
202         // make default band info
203         config.bandInfo = [
204     		{
205                 width:          "80%", 
206                 intervalUnit:   intervals[0], 
207                 intervalPixels: 70
208             },
209             {
210                 width:          "20%", 
211                 intervalUnit:   intervals[1], 
212                 intervalPixels: 100,
213                 showEventText:  false,
214                 overview: true,
215                 trackHeight:    0.4,
216                 trackGap:       0.2
217             }
218         ];
219     }
220     
221     // create the TimeMap object
222     var tm = new TimeMap(
223   		document.getElementById(config.timelineId), 
224 		document.getElementById(config.mapId),
225 		config.options);
226     
227     // create the dataset objects
228     var datasets = [], x, ds, dsOptions, dsId;
229     for (x=0; x < config.datasets.length; x++) {
230         ds = config.datasets[x];
231         dsOptions = ds.options || {};
232         dsOptions.title = ds.title || '';
233         dsOptions.theme = ds.theme;
234         dsOptions.dateParser = ds.dateParser;
235         dsId = ds.id || "ds" + x;
236         datasets[x] = tm.createDataset(dsId, dsOptions);
237         if (x > 0) {
238             // set all to the same eventSource
239             datasets[x].eventSource = datasets[0].eventSource;
240         }
241     }
242     // add a pointer to the eventSource in the TimeMap
243     tm.eventSource = datasets[0].eventSource;
244     
245     // set up timeline bands
246     var bands = [];
247     // ensure there's at least an empty eventSource
248     var eventSource = (datasets[0] && datasets[0].eventSource) || new Timeline.DefaultEventSource();
249     // check for pre-initialized bands (manually created with Timeline.createBandInfo())
250     if (config.bands) {
251         bands = config.bands;
252         // substitute dataset event source
253         for (x=0; x < bands.length; x++) {
254             // assume that these have been set up like "normal" Timeline bands:
255             // with an empty event source if events are desired, and null otherwise
256             if (bands[x].eventSource !== null) {
257                 bands[x].eventSource = eventSource;
258             }
259         }
260     }
261     // otherwise, make bands from band info
262     else {
263         for (x=0; x < config.bandInfo.length; x++) {
264             var bandInfo = config.bandInfo[x];
265             // if eventSource is explicitly set to null or false, ignore
266             if (!(('eventSource' in bandInfo) && !bandInfo.eventSource)) {
267                 bandInfo.eventSource = eventSource;
268             }
269             else {
270                 bandInfo.eventSource = null;
271             }
272             bands[x] = Timeline.createBandInfo(bandInfo);
273             if (x > 0 && TimeMap.util.TimelineVersion() == "1.2") {
274                 // set all to the same layout
275                 bands[x].eventPainter.setLayout(bands[0].eventPainter.getLayout()); 
276             }
277         }
278     }
279     // initialize timeline
280     tm.initTimeline(bands);
281     
282     // initialize load manager
283     var loadManager = TimeMap.loadManager;
284     loadManager.init(tm, config.datasets.length, config);
285     
286     // load data!
287     for (x=0; x < config.datasets.length; x++) {
288         (function(x) { // deal with closure issues
289             var data = config.datasets[x], options, type, callback, loaderClass, loader;
290             // support some older syntax
291             options = data.options || data.data || {};
292             type = data.type || options.type;
293             callback = function() { loadManager.increment() };
294             // get loader class
295             loaderClass = (typeof(type) == 'string') ? TimeMap.loaders[type] : type;
296             // load with appropriate loader
297             loader = new loaderClass(options);
298             loader.load(datasets[x], callback);
299         })(x);
300     }
301     // return timemap object for later manipulation
302     return tm;
303 };

308 /**
309  * @class Static singleton for managing multiple asynchronous loads
310  */
311 TimeMap.loadManager = new function() {
312     
313     /**
314      * Initialize (or reset) the load manager
315      *
316      * @param {TimeMap} tm          TimeMap instance
317      * @param {int} target     Number of datasets we're loading
318      * @param {Object} options      Container for optional settings:<pre>
319      *   {Function} dataLoadedFunction      Custom function replacing default completion function;
320      *                                      should take one parameter, the TimeMap object
321      *   {String/Date} scrollTo             Where to scroll the timeline when load is complete
322      *                                      Options: "earliest", "latest", "now", date string, Date
323      *   {Function} dataDisplayedFunction   Custom function to fire once data is loaded and displayed;
324      *                                      should take one parameter, the TimeMap object
325      * </pre>
326      */
327     this.init = function(tm, target, config) {
328         this.count = 0;
329         this.tm = tm;
330         this.target = target;
331         this.opts = config || {};
332     };
333     
334     /**
335      * Increment the count of loaded datasets
336      */
337     this.increment = function() {
338         this.count++;
339         if (this.count >= this.target) {
340             this.complete();
341         }
342     };
343     
344     /**
345      * Function to fire when all loads are complete. 
346      * Default behavior is to scroll to a given date (if provided) and
347      * layout the timeline.
348      */
349     this.complete = function() {
350         var tm = this.tm;
351         // custom function including timeline scrolling and layout
352         var func = this.opts.dataLoadedFunction;
353         if (func) {
354             func(tm);
355         } else {
356             var d = new Date();
357             var eventSource = this.tm.eventSource;
358             var scrollTo = this.opts.scrollTo;
359             // make sure there are events to scroll to
360             if (scrollTo && eventSource.getCount() > 0) {
361                 switch (scrollTo) {
362                     case "now":
363                         break;
364                     case "earliest":
365                         d = eventSource.getEarliestDate();
366                         break;
367                     case "latest":
368                         d = eventSource.getLatestDate();
369                         break;
370                     default:
371                         // assume it's a date, try to parse
372                         if (typeof(scrollTo) == 'string') {
373                             scrollTo = TimeMapDataset.hybridParser(scrollTo);
374                         }
375                         // either the parse worked, or it was a date to begin with
376                         if (scrollTo.constructor == Date) d = scrollTo;
377                 }
378                 this.tm.timeline.getBand(0).setCenterVisibleDate(d);
379             }
380             this.tm.timeline.layout();
381             // custom function to be called when data is loaded
382             func = this.opts.dataDisplayedFunction;
383             if (func) {
384                 func(tm);
385             }
386         }
387     };
388 };

562  * Initialize the timeline - this must happen separately to allow full control of

565  * @param {BandInfo Array} bands    Array of band information objects for timeline

568     
569     // synchronize & highlight timeline bands
570     for (var x=1; x < bands.length; x++) {
571         if (this.opts.syncBands) {
572             bands[x].syncWith = (x-1);
573         }
574         bands[x].highlight = true;

579      * @type Timeline

583     // set event listeners
584     var tm = this;
585     // update map on timeline scroll
586     this.timeline.getBand(0).addOnScrollListener(function() {
587         tm.filter("map");
588     });
589 
590     // hijack timeline popup window to open info window
591     var painter = this.timeline.getBand(0).getEventPainter().constructor;
592     painter.prototype._showBubble = function(x, y, evt) {
593         evt.item.openInfoWindow();
594     };
595     
596     // filter chain for map placemarks
597     this.addFilterChain("map", 
598         function(item) {
599             item.showPlacemark();
600         },
601         function(item) {
602             item.hidePlacemark();
603         }
604     );
605     
606     // filter: hide when item is hidden
607     this.addFilter("map", function(item) {
608         return item.visible;
609     });
610     // filter: hide when dataset is hidden
611     this.addFilter("map", function(item) {
612         return item.dataset.visible;
613     });
614     
615     // filter: hide map items depending on timeline state
616     this.addFilter("map", this.opts.mapFilter);
617     
618     // filter chain for timeline events
619     this.addFilterChain("timeline", 
620         function(item) {
621             item.showEvent();
622         },
623         function(item) {
624             item.hideEvent();
625         }
626     );
627     
628     // filter: hide when item is hidden
629     this.addFilter("timeline", function(item) {
630         return item.visible;
631     });
632     // filter: hide when dataset is hidden
633     this.addFilter("timeline", function(item) {
634         return item.dataset.visible;
635     });
636     
637     // add callback for window resize
638     var resizeTimerID = null;
639     var oTimeline = this.timeline;
640     window.onresize = function() {
641         if (resizeTimerID === null) {
642             resizeTimerID = window.setTimeout(function() {
643                 resizeTimerID = null;
644                 oTimeline.layout();
645             }, 500);
646         }
647     };
648 };

650 /**
651  * Update items, hiding or showing according to filters
652  *
653  * @param {String} fid      Filter chain to update on
654  */
655 TimeMap.prototype.filter = function(fid) {
656     var filters = this.filters[fid];
657     // if no filters exist, forget it
658     if (!filters || !filters.chain || filters.chain.length === 0) {
659         return;
660     }
661     // run items through filter
662     this.each(function(ds) {
663         ds.each(function(item) {
664             var done = false;
665             F_LOOP: while (!done) { 
666                 for (var i = filters.chain.length - 1; i >= 0; i--) {
667                     if (!filters.chain[i](item)) {
668                         // false condition
669                         filters.off(item);
670                         break F_LOOP;
671                     }
672                 }
673                 // true condition
674                 filters.on(item);
675                 done = true;
676             }
677         });
678     });
679 };

696 /**
697  * Remove a filter chain
698  *
699  * @param {String} fid      Id of the filter chain
700  */

879 /**
880  * Better Timeline Gregorian parser... shouldn't be necessary :(.
881  * Gregorian dates are years with "BC" or "AD"
882  *
883  * @param {String} s    String to parse into a Date object
884  * @return {Date}       Parsed date or null
885  */
886 TimeMapDataset.gregorianParser = function(s) {
887     if (!s) {
888         return null;
889     } else if (s instanceof Date) {
890         return s;
891     }
892     // look for BC
893     var bc = Boolean(s.match(/b\.?c\.?/i));
894     // parse - parseInt will stop at non-number characters
895     var year = parseInt(s);
896     // look for success
897     if (!isNaN(year)) {
898         // deal with BC
899         if (bc) year = 1 - year;
900         // make Date and return
901         var d = new Date(0);
902         d.setUTCFullYear(year);
903         return d;
904     }
905     else {
906         return null;
907     }
908 };

910 /**
911  * Parse date strings with a series of date parser functions, until one works. 
912  * In order:
913  * <ol>
914  *  <li>Date.parse() (so Date.js should work here, if it works with Timeline...)</li>
915  *  <li>Gregorian parser</li>
916  *  <li>The Timeline ISO 8601 parser</li>
917  * </ol>
918  *
919  * @param {String} s    String to parse into a Date object
920  * @return {Date}       Parsed date or null
921  */
922 TimeMapDataset.hybridParser = function(s) {
923     // try native date parse
924     var d = new Date(Date.parse(s));
925     if (isNaN(d)) {
926         // look for Gregorian dates
927         if (s.match(/^-?\d{1,6} ?(a\.?d\.?|b\.?c\.?e?\.?|c\.?e\.?)?$/i)) {
928             d = TimeMapDataset.gregorianParser(s);
929         }
930         // try ISO 8601 parse
931         else {
932             try {
933                 d = DT.parseIso8601DateTime(s);
934             } catch(e) {
935                 d = null;
936             }
937         }
938     }
939     // d should be a date or null
940     return d;
941 };

982  * Add one item to map and timeline. 
983  * Each item has both a timeline event and a map placemark.
984  *
985  * @param {Object} data         Data to be loaded, in the following format: <pre>
986  *      {String} title              Title of the item (visible on timeline)
987  *      {DateTime} start            Start time of the event on the timeline
988  *      {DateTime} end              End time of the event on the timeline (duration events only)
989  *      {Object} point              Data for a single-point placemark: 
990  *          {Float} lat                 Latitude of map marker
991  *          {Float} lon                 Longitude of map marker
992  *      {Array of points} polyline  Data for a polyline placemark, in format above
993  *      {Array of points} polygon   Data for a polygon placemark, in format above
994  *      {Object} overlay            Data for a ground overlay:
995  *          {String} image              URL of image to overlay
996  *          {Float} north               Northern latitude of the overlay
997  *          {Float} south               Southern latitude of the overlay
998  *          {Float} east                Eastern longitude of the overlay
999  *          {Float} west                Western longitude of the overlay
1000  *      {Object} options            Optional arguments to be passed to the TimeMapItem (@see TimeMapItem)
1001  * </pre>
1002  * @param {Function} transform  If data is not in the above format, transformation function to make it so
1003  * @return {TimeMapItem}        The created item (for convenience, as it's already be added)
1004  * @see TimeMapItem
1005  */
1006 TimeMapDataset.prototype.loadItem = function(data, transform) {
1007     // apply transformation, if any
1008     if (transform !== undefined) {
1009         data = transform(data);
1010     }
1011     // transform functions can return a null value to skip a datum in the set
1012     if (data === null) {
1013         return;
1014     }
1015     
1016     var options = data.options || {},
1017         tm = this.timemap,
1018         dstheme = this.opts.theme,
1019         theme = options.theme || dstheme;
1020     
1021     // event icon path overrides custom themes
1022     options.eventIconPath = options.eventIconPath || dstheme.eventIconPath;
1023     // get configured theme
1024     theme = TimeMapTheme.create(theme, options);
1025     
1026     // create timeline event
1027     var parser = this.opts.dateParser, start = data.start, end = data.end, instant;
1028     start = (start === undefined||start === "") ? null : parser(start);
1029     end = (end === undefined||end === "") ? null : parser(end);
1030     instant = (end === undefined);
1031     var eventIcon = theme.eventIcon,
1032         title = data.title,
1033         // allow event-less placemarks - these will be always present on map
1034         event = null;
1035     if (start !== null) {
1036         var eventClass = Timeline.DefaultEventSource.Event;
1037         if (TimeMap.util.TimelineVersion() == "1.2") {
1038             // attributes by parameter
1039             event = new eventClass(start, end, null, null,
1040                 instant, title, null, null, null, eventIcon, theme.eventColor, 
1041                 theme.eventTextColor);
1042         } else {
1043             var textColor = theme.eventTextColor;
1044             if (!textColor) {
1045                 // tweak to show old-style events
1046                 textColor = (theme.classicTape && !instant) ? '#FFFFFF' : '#000000';
1047             }
1048             // attributes in object
1049             event = new eventClass({
1050                 "start": start,
1051                 "end": end,
1052                 "instant": instant,
1053                 "text": title,
1054                 "icon": eventIcon,
1055                 "color": theme.eventColor,
1056                 "textColor": textColor
1057             });
1058         }
1059     }
1060     
1061     // set the icon, if any, outside the closure
1062     var markerIcon = theme.icon,
1063         bounds = tm.mapBounds; // save some bytes
1064     
1065     
1066     // internal function: create map placemark
1067     // takes a data object (could be full data, could be just placemark)
1068     // returns an object with {placemark, type, point}
1069     var createPlacemark = function(pdata) {
1070         var placemark = null, type = "", point = null;
1071         // point placemark
1072         if ("point" in pdata) {
1073             point = new GLatLng(
1074                 parseFloat(pdata.point.lat), 
1075                 parseFloat(pdata.point.lon)
1076             );
1077             // add point to visible map bounds
1078             if (tm.opts.centerOnItems) {
1079                 bounds.extend(point);
1080             }
1081             placemark = new GMarker(point, { icon: markerIcon });
1082             type = "marker";
1083             point = placemark.getLatLng();
1084         }
1085         // polyline and polygon placemarks
1086         else if ("polyline" in pdata || "polygon" in pdata) {
1087             var points = [], line;
1088             if ("polyline" in pdata) {
1089                 line = pdata.polyline;
1090             } else {
1091                 line = pdata.polygon;
1092             }
1093             for (var x=0; x<line.length; x++) {
1094                 point = new GLatLng(
1095                     parseFloat(line[x].lat), 
1096                     parseFloat(line[x].lon)
1097                 );
1098                 points.push(point);
1099                 // add point to visible map bounds
1100                 if (tm.opts.centerOnItems) {
1101                     bounds.extend(point);
1102                 }
1103             }
1104             if ("polyline" in pdata) {
1105                 placemark = new GPolyline(points, 
1106                                           theme.lineColor, 
1107                                           theme.lineWeight,
1108                                           theme.lineOpacity);
1109                 type = "polyline";
1110                 point = placemark.getVertex(Math.floor(placemark.getVertexCount()/2));
1111             } else {
1112                 placemark = new GPolygon(points, 
1113                                          theme.polygonLineColor, 
1114                                          theme.polygonLineWeight,
1115                                          theme.polygonLineOpacity,
1116                                          theme.fillColor,
1117                                          theme.fillOpacity);
1118                 type = "polygon";
1119                 point = placemark.getBounds().getCenter();
1120             }
1121         } 
1122         // ground overlay placemark
1123         else if ("overlay" in pdata) {
1124             var sw = new GLatLng(
1125                 parseFloat(pdata.overlay.south), 
1126                 parseFloat(pdata.overlay.west)
1127             );
1128             var ne = new GLatLng(
1129                 parseFloat(pdata.overlay.north), 
1130                 parseFloat(pdata.overlay.east)
1131             );
1132             // add to visible bounds
1133             if (tm.opts.centerOnItems) {
1134                 bounds.extend(sw);
1135                 bounds.extend(ne);
1136             }
1137             // create overlay
1138             var overlayBounds = new GLatLngBounds(sw, ne);
1139             placemark = new GGroundOverlay(pdata.overlay.image, overlayBounds);
1140             type = "overlay";
1141             point = overlayBounds.getCenter();
1142         }
1143         return {
1144             "placemark": placemark,
1145             "type": type,
1146             "point": point
1147         };
1148     };
1149     
1150     // create placemark or placemarks
1151     var placemark = [], pdataArr = [], pdata = null, type = "", point = null, i;
1152     // array of placemark objects
1153     if ("placemarks" in data) {
1154         pdataArr = data.placemarks;
1155     } else {
1156         // we have one or more single placemarks
1157         var types = ["point", "polyline", "polygon", "overlay"];
1158         for (i=0; i<types.length; i++) {
1159             if (types[i] in data) {
1160                 pdata = {};
1161                 pdata[types[i]] = data[types[i]];
1162                 pdataArr.push(pdata);
1163             }
1164         }
1165     }
1166     if (pdataArr) {
1167         for (i=0; i<pdataArr.length; i++) {
1168             // create the placemark
1169             var p = createPlacemark(pdataArr[i]);
1170             // take the first point and type as a default
1171             point = point || p.point;
1172             type = type || p.type;
1173             placemark.push(p.placemark);
1174         }
1175     }
1176     // override type for arrays
1177     if (placemark.length > 1) {
1178         type = "array";
1179     }
1180     
1181     options.title = title;
1182     options.type = type || "none";
1183     options.theme = theme;
1184     // check for custom infoPoint and convert to GLatLng
1185     if (options.infoPoint) {
1186         options.infoPoint = new GLatLng(
1187             parseFloat(options.infoPoint.lat), 
1188             parseFloat(options.infoPoint.lon)
1189         );
1190     } else {
1191         options.infoPoint = point;
1192     }
1193     
1194     // create item and cross-references
1195     var item = new TimeMapItem(placemark, event, this, options);
1196     // add event if it exists
1197     if (event !== null) {
1198         event.item = item;
1199         this.eventSource.add(event);
1200     }
1201     // add placemark(s) if any exist
1202     if (placemark.length > 0) {
1203         for (i=0; i<placemark.length; i++) {
1204             placemark[i].item = item;
1205             // add listener to make placemark open when event is clicked
1206             GEvent.addListener(placemark[i], "click", function() {
1207                 item.openInfoWindow();
1208             });
1209             // add placemark and event to map and timeline
1210             tm.map.addOverlay(placemark[i]);
1211             // hide placemarks until the next refresh
1212             placemark[i].hide();
1213         }
1214     }
1215     // add the item to the dataset
1216     this.items.push(item);
1217     // return the item object
1218     return item;
1219 };

1787 /**
1788  * Make TimeMap.init()-style points from a GLatLng, array, or string
1789  *
1790  * @param {Object} coords       GLatLng, array, or string to convert
1791  * @param {Boolean} reversed    Whether the points are KML-style lon/lat, rather than lat/lon
1792  * @return {Object}             TimeMap.init()-style point 
1793  */
1794 TimeMap.util.makePoint = function(coords, reversed) {
1795     var latlon = null, 
1796         trim = TimeMap.util.trim;
1797     // GLatLng
1798     if (coords.lat && coords.lng) {
1799         latlon = [coords.lat(), coords.lng()];
1800     }
1801     // array of coordinates
1802     if (TimeMap.util.isArray(coords)) {
1803         latlon = coords;
1804     }
1805     // string
1806     if (latlon === null) {
1807         // trim extra whitespace
1808         coords = trim(coords);
1809         if (coords.indexOf(',') > -1) {
1810             // split on commas
1811             latlon = coords.split(",");
1812         } else {
1813             // split on whitespace
1814             latlon = coords.split(/[\r\n\f ]+/);
1815         }
1816     }
1817     // deal with extra coordinates (i.e. KML altitude)
1818     if (latlon.length > 2) latlon = latlon.slice(0, 2);
1819     // deal with backwards (i.e. KML-style) coordinates
1820     if (reversed) latlon.reverse();
1821     return {
1822         "lat": trim(latlon[0]),
1823         "lon": trim(latlon[1])
1824     };
1825 };

1827 /**
1828  * Make TimeMap.init()-style polyline/polygons from a whitespace-delimited
1829  * string of coordinates (such as those in GeoRSS and KML).
1830  * XXX: Any reason for this to take arrays of GLatLngs as well?
1831  *
1832  * @param {Object} coords       String to convert
1833  * @param {Boolean} reversed    Whether the points are KML-style lon/lat, rather than lat/lon
1834  * @return {Object}             Formated coordinate array
1835  */
1836 TimeMap.util.makePoly = function(coords, reversed) {
1837     var poly = [], latlon;
1838     var coordArr = TimeMap.util.trim(coords).split(/[\r\n\f ]+/);
1839     if (coordArr.length == 0) return [];
1840     // loop through coordinates
1841     for (var x=0; x<coordArr.length; x++) {
1842         latlon = (coordArr[x].indexOf(',') > 0) ?
1843             // comma-separated coordinates (KML-style lon/lat)
1844             coordArr[x].split(",") :
1845             // space-separated coordinates - increment to step by 2s
1846             [coordArr[x], coordArr[++x]];
1847         // deal with extra coordinates (i.e. KML altitude)
1848         if (latlon.length > 2) latlon = latlon.slice(0, 2);
1849         // deal with backwards (i.e. KML-style) coordinates
1850         if (reversed) latlon.reverse();
1851         poly.push({
1852             "lat": latlon[0],
1853             "lon": latlon[1]
1854         });
1855     }
1856     return poly;
1857 }

1859 /**
1860  * Format a date as an ISO 8601 string
1861  *
1862  * @param {Date} d          Date to format
1863  * @param {int} precision   Optional precision indicator:
1864  *                              3 (default): Show full date and time
1865  *                              2: Show full date and time, omitting seconds
1866  *                              1: Show date only
1867  * @return {String}         Formatted string
1868  */
1869 TimeMap.util.formatDate = function(d, precision) {
1870     // default to high precision
1871     precision = precision || 3;
1872     var str = "";
1873     if (d) {
1874         // check for date.js support
1875         if (d.toISOString) {
1876             return d.toISOString();
1877         }
1878         // otherwise, build ISO 8601 string
1879         var pad = function(num) {
1880             return ((num < 10) ? "0" : "") + num;
1881         };
1882         var yyyy = d.getUTCFullYear(),
1883             mo = d.getUTCMonth(),
1884             dd = d.getUTCDate();
1885         str += yyyy + '-' + pad(mo + 1 ) + '-' + pad(dd);
1886         // show time if top interval less than a week
1887         if (precision > 1) {
1888             var hh = d.getUTCHours(),
1889                 mm = d.getUTCMinutes(),
1890                 ss = d.getUTCSeconds();
1891             str += 'T' + pad(hh) + ':' + pad(mm);
1892             // show seconds if the interval is less than a day
1893             if (precision > 2) {
1894                 str += pad(ss);
1895             }
1896             str += 'Z';
1897         }
1898     }
1899     return str;
1900 };
