 21 /**
 22  * Delete all datasets, clearing them from map and timeline
 23  */
 24 TimeMap.prototype.clear = function() {
 25     this.each(function(ds) {
 26         ds.clear();
 27     });
 28     this.datasets = [];
 29 };

 31 /**
 32  * Delete one dataset, clearing it from map and timeline
 33  *
 34  * @param id    Id of dataset to delete
 35  */
 36 TimeMap.prototype.deleteDataset = function(id) {
 37     this.datasets[id].clear();
 38     delete this.datasets[id];
 39 };

 87 /**
 88  * Change the default map type
 89  *
 90  * @param {String or Object} mapType   New map type If string, looks up in TimeMap.mapTypes.
 91  */
 92 TimeMap.prototype.changeMapType = function (mapType) {
 93     // check for no change
 94     if (mapType == this.opts.mapType) {
 95         return;
 96     }
 97     // look for mapType
 98     if (typeof(mapType) == 'string') {
 99         mapType = TimeMap.mapTypes[mapType];
100     }
101     // no mapType specified
102     if (!mapType) {
103         return;
104     }
105     // change it
106     this.opts.mapType = mapType;
107     this.map.setMapType(mapType);
108 };

114 /**
115  * Refresh the timeline, maintaining the current date
116  */
117 TimeMap.prototype.refreshTimeline = function () {
118     var topband = this.timeline.getBand(0);
119     var centerDate = topband.getCenterVisibleDate();
120     if (TimeMap.util.TimelineVersion() == "1.2") {
121         topband.getEventPainter().getLayout()._laidout = false;
122     }
123     this.timeline.layout();
124     topband.setCenterVisibleDate(centerDate);
125 };
126 
127 /**
128  * Change the intervals on the timeline.
129  *
130  * @param {String or Array} intervals   New intervals. If string, looks up in TimeMap.intervals.
131  */
132 TimeMap.prototype.changeTimeIntervals = function (intervals) {
133     // check for no change
134     if (intervals == this.opts.bandIntervals) {
135         return;
136     }
137     // look for intervals
138     if (typeof(intervals) == 'string') {
139         intervals = TimeMap.intervals[intervals];
140     }
141     // no intervals specified
142     if (!intervals) {
143         return;
144     }
145     this.opts.bandIntervals = intervals;
146     // internal function - change band interval
147     var changeInterval = function(band, interval) {
148         band.getEther()._interval = Timeline.DateTime.gregorianUnitLengths[interval];
149         band.getEtherPainter()._unit = interval;
150     };
151     // grab date
152     var topband = this.timeline.getBand(0);
153     var centerDate = topband.getCenterVisibleDate();
154     // change interval for each band
155     for (var x=0; x<this.timeline.getBandCount(); x++) {
156         changeInterval(this.timeline.getBand(x), intervals[x]);
157     }
158     // re-layout timeline
159     topband.getEventPainter().getLayout()._laidout = false;
160     this.timeline.layout();
161     topband.setCenterVisibleDate(centerDate);
162 };
163  
164 /**
165  * Scrolls the timeline the number of years passed (negative numbers scroll it back)
166  * XXX: This should probably handle other intervals as well...
167  *
168  * @param {int} years    Number of years to scroll the timeline
169 */
170 TimeMap.prototype.scrollTimeline = function (years) {
171  	var topband = this.timeline.getBand(0);
172  	var centerDate = topband.getCenterVisibleDate();
173  	var centerYear = centerDate.getFullYear() + parseFloat(years);
174  	centerDate.setFullYear(centerYear);
175  	topband.setCenterVisibleDate(centerDate);
176 };

183 /**
184  * Delete all items, clearing them from map and timeline
185  */
186 TimeMapDataset.prototype.clear = function() {
187     this.each(function(item) {
188         item.clear();
189     });
190     this.items = [];
191     this.timemap.timeline.layout();
192 };
193 
194 /**
195  * Delete one item, clearing it from map and timeline
196  * 
197  * @param item      Item to delete
198  */
199 TimeMapDataset.prototype.deleteItem = function(item) {
200     for (var x=0; x < this.items.length; x++) {
201         if (this.items[x] == item) {
202             item.clear();
203             this.items.splice(x, 1);
204             break;
205         }
206     }
207     this.timemap.timeline.layout();
208 };

210 /**
211  * Show dataset
212  */
213 TimeMapDataset.prototype.show = function() {
214     if (!this.visible) {
215         this.visible = true;
216         this.timemap.filter("map");
217         this.timemap.filter("timeline");
218         this.timemap.timeline.layout();
219     }
220 };
221 
222 /**
223  * Hide dataset
224  */
225 TimeMapDataset.prototype.hide = function() {
226     if (this.visible) {
227         this.visible = false;
228         this.timemap.filter("map");
229         this.timemap.filter("timeline");
230         this.timemap.timeline.layout();
231     }
232 };

270 /**
271  * Delete placemark from map and event from timeline
272  */
273 TimeMapItem.prototype.clear = function() {
274     if (this.event) {
275         // this is just ridiculous
276         this.dataset.timemap.timeline.getBand(0)
277             .getEventSource()._events._events.remove(this.event);
278     }
279     if (this.placemark) {
280         this.hidePlacemark();
281         var f = function(p) {
282             try {
283                 this.map.removeOverlay(p);
284             } catch(e) {}
285         };
286         if (this.getType() == "array") {
287             for (var i=0; i<this.placemark.length; i++) {
288                 f(this.placemark[i]);
289             }
290         } else {
291             f(this.placemark);
292         }
293     }
294     this.event = this.placemark = null;
295 };

297  /**
298  * Create a new event for the item.
299  * 
300  * @param (Date) s      Start date for the event
301  * @param (Date) e      (Optional) End date for the event
302  */
303 TimeMapItem.prototype.createEvent = function(s, e) {
304     var instant = (e === undefined);
305     var title = this.getTitle();
306     // create event
307     var event = new Timeline.DefaultEventSource.Event(s, e, null, null, instant, title, 
308         null, null, null, this.opts.theme.eventIcon, this.opts.theme.eventColor, null);
309     // add references
310     event.item = this;
311     this.event = event;
312     this.dataset.eventSource.add(event);
313 };

315  /**
316  * Change the theme for an item
317  *
318  * @param theme   New theme settings
319  */
320  TimeMapItem.prototype.changeTheme = function(newTheme) {
321     this.opts.theme = newTheme;
322     // change placemark
323     if (this.placemark) {
324         // internal function - takes type, placemark
325         var changePlacemark = function(pm, type, theme) {
326             type = type || TimeMap.util.getPlacemarkType(pm);
327             switch (type) {
328                 case "marker":
329                     pm.setImage(theme.icon.image);
330                     break;
331                 case "polygon":
332                     pm.setFillStyle({
333                         'color': newTheme.fillColor,
334                         'opacity': newTheme.fillOpacity
335                     });
336                     // no break to get stroke style too
337                 case "polyline":
338                     pm.setStrokeStyle({
339                         'color': newTheme.lineColor,
340                         'weight': newTheme.lineWeight,
341                         'opacity': newTheme.lineOpacity
342                     });
343                     break;
344             }
345         };
346         if (this.getType() == 'array') {
347             for (var i=0; i<this.placemark.length; i++) {
348                 changePlacemark(this.placemark[i], false, newTheme);
349             }
350         } else {
351             changePlacemark(this.placemark, this.getType(), newTheme);
352         }
353     }
354     // change event
355     if (this.event) {
356         this.event._color = newTheme.eventColor;
357         this.event._icon = newTheme.eventIcon;
358     }
359 };

361 /** 
362  * Find the next item chronologically
363  *
364  * @param {Boolean} inDataset   Whether to only look in this item's dataset
365  * @return {TimeMapItem}        Next item, if any
366  */
367 TimeMapItem.prototype.getNext = function(inDataset) {
368     if (!this.event) {
369         return;
370     }
371     var eventsource = this.dataset.timemap.timeline.getBand(0).getEventSource();
372     // iterator dates are non-inclusive, hence the juggle here
373     var i = eventsource.getEventIterator(this.event.getStart(), 
374         new Date(eventsource.getLatestDate().getTime() + 1));
375     var next = null;
376     while (next === null) {
377         if (i.hasNext()) {
378             next = i.next().item;
379             if (inDataset && next.dataset != this.dataset) {
380                 next = null;
381             }
382         } else {
383             break;
384         }
385     }
386     return next;
387 };
